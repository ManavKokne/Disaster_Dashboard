-- PostgreSQL schema and seed script for disaster.tweets

CREATE TABLE IF NOT EXISTS tweets (
    id SERIAL PRIMARY KEY,
    content TEXT NOT NULL,
    location VARCHAR(255) NOT NULL,
    request_type VARCHAR(100),
    urgency VARCHAR(20) NOT NULL CHECK (urgency IN ('urgent', 'non-urgent')),

    -- Status fields
    is_acknowledged BOOLEAN NOT NULL DEFAULT FALSE,
    is_resolved BOOLEAN NOT NULL DEFAULT FALSE,
    resolved_at TIMESTAMP NULL,
    is_closed BOOLEAN NOT NULL DEFAULT FALSE,
    closed_at TIMESTAMP NULL,

    -- Geolocation fields (filled by async worker)
    latitude NUMERIC(10, 8) NULL,
    longitude NUMERIC(11, 8) NULL,
    geocode_status VARCHAR(20) NOT NULL DEFAULT 'pending'
        CHECK (
            geocode_status = 'pending'
            OR geocode_status = 'done'
            OR geocode_status = 'failed'
            OR geocode_status LIKE 'failed_%'
        ),

    -- Audit timestamps
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX IF NOT EXISTS idx_active_polling ON tweets (is_closed, created_at);
CREATE INDEX IF NOT EXISTS idx_urgency ON tweets (urgency);
CREATE INDEX IF NOT EXISTS idx_geocode ON tweets (geocode_status);
CREATE INDEX IF NOT EXISTS idx_location ON tweets (latitude, longitude);

-- Keep updated_at current on every row update.
CREATE OR REPLACE FUNCTION set_updated_at()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS tweets_set_updated_at ON tweets;
CREATE TRIGGER tweets_set_updated_at
BEFORE UPDATE ON tweets
FOR EACH ROW
EXECUTE FUNCTION set_updated_at();
